            <h2 style="margin-top:26px;">How to read these charts</h2>

            <ul class="tips">

                <li><strong>Lowest earners often repay the least overall.</strong> If your salary stays below the repayment threshold, you may pay little or nothing - and any remaining balance is written off at the end of the term.</li>

                <li><strong>The very highest earners also come out well.</strong> They clear the whole balance quickly, so interest has less time to compound and they stop paying once the loan is gone.</li>

                <li><strong>Middle earners frequently pay the most.</strong> They earn enough to make sizeable repayments, but not enough to clear the balance early - so interest keeps building and they pay for most (or all) of the term.</li>

                <li><strong>It behaves more like a graduate tax than a normal loan.</strong> Repayments are a percentage of income above the threshold, not a fixed instalment, so what you pay tracks your salary rather than the size of the debt.</li>

                <li><strong>Overpaying only helps if you'd actually clear the balance.</strong> If your projection shows a write-off, extra voluntary payments may just hand money to the loan you'd never have repaid anyway.</li>

                <li><strong>Watch the real (inflation-adjusted) line.</strong> A large future balance is worth less in today's money - the gap between the cash and real lines shows how much inflation quietly erodes the debt over time.</li>

            </ul>

            const p = PLANS[key];

        // ---- Core projection ----

        function project(startSalary) {

            const balance0 = +inputs.balance.value;

            let salary = startSalary !== null ? startSalary : +inputs.salary.value;

            const growth = +inputs.growth.value / 100;

            let threshold = +inputs.threshold.value;

            const interest = +inputs.interest.value / 100;

            const inflation = +inputs.inflation.value / 100;

            const writeoff = +inputs.writeoff.value;


            const years = [0];

            const nominal = [balance0];

            const real = [balance0];


            let bal = balance0;

            let totalRepaidNom = 0;

            let totalRepaidReal = 0;

            let clearedYear = null;


            for (let y = 1; y <= writeoff; y++) {

                // Interest accrues on the outstanding balance.

                bal *= (1 + interest);

                // Annual repayment = repayRate of income above the (inflation-rising) threshold.

                const repayment = Math.max(0, (salary - threshold)) * repayRate;

                const actualRepay = Math.min(repayment, bal);

                bal -= actualRepay;


                totalRepaidNom += actualRepay;

                totalRepaidReal += actualRepay / Math.pow(1 + inflation, y);


                years.push(y);

                nominal.push(Math.max(0, bal));

                real.push(Math.max(0, bal) / Math.pow(1 + inflation, y));


                if (bal <= 0 && clearedYear === null) { clearedYear = y; }


                // Grow salary and uprate the threshold with inflation for next year.

                salary *= (1 + growth);

                threshold *= (1 + inflation);

            }


            const writtenOff = clearedYear === null ? Math.max(0, bal) : 0;

            return { years, nominal, real, totalRepaidNom, totalRepaidReal, clearedYear, writtenOff, writeoff };

        }
